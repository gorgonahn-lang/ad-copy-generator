export default async function handler(req, res) {
  // CORS 허용 (중요!)
  res.setHeader('Access-Control-Allow-Origin', '*');
  res.setHeader('Access-Control-Allow-Methods', 'GET, OPTIONS');
  res.setHeader('Access-Control-Allow-Headers', 'Content-Type');
  
  if (req.method === 'OPTIONS') {
    return res.status(200).end();
  }

  const { keyword, clientId, clientSecret } = req.query;

  if (!keyword || !clientId || !clientSecret) {
    return res.status(400).json({ error: '필수 파라미터 누락' });
  }

  try {
    // 네이버 검색광고 API 호출
    const apiUrl = `https://api.naver.com/keywordstool?hintKeywords=${encodeURIComponent(keyword)}&showDetail=1`;
    
    const response = await fetch(apiUrl, {
      method: 'GET',
      headers: {
        'X-Naver-Client-Id': clientId,
        'X-Naver-Client-Secret': clientSecret,
        'Content-Type': 'application/json'
      }
    });

    if (!response.ok) {
      throw new Error('네이버 API 호출 실패');
    }

    const data = await response.json();

    // 검색량 데이터 파싱
    if (data.keywordList && data.keywordList.length > 0) {
      const keywordInfo = data.keywordList[0];
      return res.status(200).json({
        mobile_search: keywordInfo.monthlyMobileQcCnt || 'N/A',
        pc_search: keywordInfo.monthlyPcQcCnt || 'N/A',
        competition: keywordInfo.compIdx || 'N/A',
        source: '네이버 API'
      });
    }

    return res.status(404).json({ error: '데이터 없음' });

  } catch (error) {
    console.error('API 오류:', error);
    return res.status(500).json({ 
      error: 'API 호출 중 오류 발생',
      message: error.message 
    });
  }
}
